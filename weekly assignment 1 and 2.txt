import java.util.Scanner;


public class TriangleArea {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        
        // Take base and height in cm
        System.out.print("Enter base in cm: ");
        double baseCm = input.nextDouble();
        
        System.out.print("Enter height in cm: ");
        double heightCm = input.nextDouble();
        
        // Calculate area in square cm
        double areaSqCm = 0.5 * baseCm * heightCm;
        
        // Convert to square inches (1 inch = 2.54 cm, so 1 sq inch = 2.54 * 2.54 sq cm)
        double cmPerInch = 2.54;
        double areaSqInches = areaSqCm / (cmPerInch * cmPerInch);
        
        // Output the result
        System.out.println("The Area of the triangle in sq in is " + areaSqInches + " and sq cm is " + areaSqCm);
        
        input.close();
    }
}

import java.util.Scanner;


public class CelsiusToFahrenheit {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        
        System.out.print("Enter temperature in Celsius: ");
        double celsius = input.nextDouble();
        
        // Use formula: (°C * 9/5) + 32 = °F
        double fahrenheitResult = (celsius * 9 / 5) + 32;
        
        System.out.println("The " + celsius + " celsius is " + fahrenheitResult + " fahrenheit");
        
        input.close();
    }
}

import java.util.Scanner;


public class FahrenheitToCelsius {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        
        System.out.print("Enter temperature in Fahrenheit: ");
        double fahrenheit = input.nextDouble();
        
        // Use formula: (°F - 32) * 5/9 = °C
        double celsiusResult = (fahrenheit - 32) * 5 / 9;
        
        System.out.println("The " + fahrenheit + " fahrenheit is " + celsiusResult + " celsius");
        
        input.close();
    }
}
